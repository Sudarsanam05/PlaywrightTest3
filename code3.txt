function categorizeError(message) {
  
  const normalized = message
    .trim()
    .toLowerCase()
    .replace(/\s+/g, " ");

  let category;

  if (normalized.includes("timeout")) {
    category = "TIMEOUT";
  } else if (normalized.includes("locator")) {
    category = "LOCATOR";
  } else {
    category = "GENERAL";
  }

  console.log("Normalized Message:", normalized);
  console.log("Category:", category);
}
